## Example Request

```bash
curl -X POST https://api.zinc.io/v1/orders \
  -u <client_token>: \
  -H "Content-Type: application/json" \
  -d '{
    "retailer": "amazon",
    "products": [
      {
        "product_id": "B000NPYY04",
        "quantity": 1
      }
    ],
    "shipping_address": {
      "first_name": "John",
      "last_name": "Doe",
      "address_line1": "123 Main St",
      "city": "New York",
      "state": "NY",
      "zip_code": "10001",
      "country": "US"
    },
    "billing_address": {
      "first_name": "John",
      "last_name": "Doe",
      "address_line1": "123 Main St",
      "city": "New York",
      "state": "NY",
      "zip_code": "10001",
      "country": "US"
    },
    "payment_method": {
      "name_on_card": "John Doe",
      "number": "4111111111111111",
      "security_code": "123",
      "expiration_month": 12,
      "expiration_year": 2025
    },
    "retailer_credentials": {
      "email": "user@example.com",
      "password": "password123"
    },
    "max_price": 5000,
    "shipping_method": "cheapest"
  }'
```
